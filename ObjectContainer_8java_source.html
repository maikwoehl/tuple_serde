111 private Object invokeMethodUnsafe() throws IllegalAccessException, IllegalArgumentException, InvocationTargetException {
112 /* Get method name of invoking method from stack trace. As this is the last one, and we have a
113 * wrapper method, it should be method 2 in the list.
114 */
115 StackTraceElement[] stacktrace = Thread.currentThread().getStackTrace();
116 StackTraceElement e = stacktrace[METHOD_ON_STACK_TRACE];

118 LOGGER.info("MethodName: " + methodName);

121 Method callerMethod = null;

125 LOGGER.debug("Scope class: " + this.getClass().getName());
126 LOGGER.debug("Parent class: " + this.getClass().getSuperclass().getName());
127 callerMethod = this.getClass().getDeclaredMethod(methodName, (Class<?>[]) null);
128 LOGGER.debug("Caller: " + callerMethod.getName());
129 } catch (NoSuchMethodException | SecurityException e1) {
130 // TODO Auto-generated catch block
131 e1.printStackTrace();
132 }
133
134 if (callerMethod.isAnnotationPresent(MemberName.class)) {
135 memberName = (callerMethod.getAnnotation(MemberName.class)).value()[0];
136 LOGGER.debug("MemberName: " + memberName);
137 }
138
139 try {
140 String getterName = getGetterNameOfMember(memberName);
141 LOGGER.debug("Getter: " + getterName);
142 LOGGER.debug("Methods in base: ");
143 for (Method method : this.getClass().getSuperclass().getDeclaredMethods()) {
144 LOGGER.debug(" - " + method.getName());
145 }
146 getterMethod = this.getClass().getSuperclass().getDeclaredMethod(getterName, (Class<?>[]) null);
147 } catch (NoSuchMethodException | SecurityException e1) {

156 memberField.setAccessible(true);
