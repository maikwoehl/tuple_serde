161 private Object invokeMethodUnsafe() throws IllegalAccessException, IllegalArgumentException, InvocationTargetException {
162 /* Get method name of invoking method from stack trace. As this is the last one, and we have a
163 * wrapper method, it should be method 2 in the list.
164 */
165 StackTraceElement[] stacktrace = Thread.currentThread().getStackTrace();
166 StackTraceElement e = stacktrace[METHOD_ON_STACK_TRACE];
167 String methodName = e.getMethodName();
168 LOGGER.info("MethodName: " + methodName);
169
170 String memberName = null;
171 Method callerMethod = null;
172 Method getterMethod = null;
173
174 try {
175 LOGGER.debug("Scope class: " + this.getClass().getName());
176 LOGGER.debug("Parent class: " + this.getClass().getSuperclass().getName());
177 callerMethod = this.getClass().getDeclaredMethod(methodName, (Class<?>[]) null);
178 LOGGER.debug("Caller: " + callerMethod.getName());
179 } catch (NoSuchMethodException | SecurityException e1) {
180 // TODO Auto-generated catch block
181 e1.printStackTrace();
182 }
183
184 if (callerMethod.isAnnotationPresent(MemberName.class)) {
185 memberName = (callerMethod.getAnnotation(MemberName.class)).value()[0];
186 LOGGER.debug("MemberName: " + memberName);
187 }
188
189 try {
190 String getterName = getGetterNameOfMember(memberName);
191 LOGGER.debug("Getter: " + getterName);
192 LOGGER.debug("Methods in base: ");
193 for (Method method : this.getClass().getSuperclass().getDeclaredMethods()) {
194 LOGGER.debug(" - " + method.getName());
195 }
196 getterMethod = this.getClass().getSuperclass().getDeclaredMethod(getterName, (Class<?>[]) null);
197 } catch (NoSuchMethodException | SecurityException e1) {
198 // TODO Auto-generated catch block

200 }

202 LOGGER.info("Execute: " + getterMethod.getName());

204 try {
205 Field memberField = this.getClass().getDeclaredField(memberName);
206 memberField.setAccessible(true);
207 memberField.set(this, returnValue);
208 } catch (NoSuchFieldException | SecurityException e1) {
